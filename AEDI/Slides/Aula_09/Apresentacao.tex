\section{Comando Condicional if}
\begin{frame}
  \frametitle{Comando Condicional if}
\begin{itemize}
 \item Permite decidir se um determinado bloco de comandos deve ou não ser executado, a partir do resultado de uma expressão lógica.
 \item O comando if serve para alterar o fluxo de execução de um programa baseado no valor, verdadeiro ou falso, de uma expressão lógica.
\end{itemize}
\end{frame}

\section{Comando Condicional if/else}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}
  \frametitle{Comando Condicional if/else}
\begin{itemize}
 \item Quando a expressão \underline{não} é verdadeira, também é possível exercutar comandos.
 \item O comando {\bf else}, sempre vem após o comando {\bf if}, serve para executar comandos caso a expressão seja falsa.
\end{itemize}
\end{frame}

\item Caso positivo, também deve informar há quantos anos a pessoa já tem direito a carteira de habilitação.

#include<stdio.h>
int main() {
  // Declaracao de variaveis
  int idade = 0;
  // Leitura dos dados
  printf("Digite a sua idade:");
  scanf("%d",&idade);
  // Verificacao
  if (idade>=18) {
      printf("Voce pode obter CNH\n");
      printf("Voce jah tem direito a %d anos.", idade-18);
  }
  else {
      printf("Voce ainda nao pode obter CNH\n");

\begin{frame}\frametitle{Operadores Lógicos}
\begin{itemize}
 \item Os operadores lógicos unem duas ou mais expressões em uma única expressão.
 \item São três os operadores lógicos: {\bf AND}, {\bf OR} e {\bf NOT}.
\end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Comando Condicional if/else}
{\bf Exemplo:} escreva um programa que leia a idade de uma pessoa e informe se o voto é facultativo, obrigatório ou se a pessoa não pode votar. Lembre-se que o voto é obrigatório para os cidadãos brasileiros maiores de 18 e menores de 70 anos e facultativo para quem tem 16 e 17 anos e para os maiores de 70 anos.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]
  \frametitle{Comando Condicional if}
\begin{lstlisting}[basicstyle=\tiny]
#include<stdio.h>
int main() {
  	// Declaracao de variaveis
	int idade = 0;
	// Entrada dos dados
	printf("Digite a sua idade:");
  	scanf("%d",&idade);
	// Verificacao se voto obrigatorio
  	if (idade>=18 && idade <=70) {
      	printf("Voto obrigatorio\n");
  	}
  	// Verificacao se voto facultativo
  	else 
  		if (idade>70 || (idade < 18 && idade >=16)) {
      		printf("Voto facultativo\n");
		}
	  	// Verificacao se nao pode votar
	  	else {
      		printf("Voce nao pode votar\n");
		}
  	return 0;
}
\end{lstlisting}
\end{frame}
