\section{Comando For}

\begin{frame}
  \frametitle{Comando For}
  \begin{itemize}
   \item O comando {\bf for} é útil quando se conhece previamente o número exato de vezes que se deseja executar um determinado conjunto de comandos.
   \item Sua estrutura possui mecanismos para contar o número de vezes que o corpo do laço é executado.
  \end{itemize}
\end{frame}


\begin{frame}[fragile]
  \frametitle{Comando While}
A forma do comando {\bf while} na linguagem C é a seguinte:
\begin{lstlisting}
inicializacao;

while ( condicao ) {
  conjunto de instrucoes; // Comandos a serem repetidos
  
  incremento;
}
\end{lstlisting}
\end{frame}


\begin{frame}[fragile]
  \frametitle{Comando For}
A forma do comando {\bf for} na linguagem C é a seguinte:
\begin{lstlisting}
for (inicializacao; condicao; incremento) {
  conjunto de instrucoes; // Comandos a serem repetidos
} 
\end{lstlisting}
\end{frame}

\section{Comando Break}

\begin{frame}[fragile]
  \frametitle{Comando Break}
\begin{itemize}
 \item Usando o comando {\bf break}, podemos \underline{interromper} um {\it loop} mesmo que a condição ainda seja verdadeira.
 \item Pode ser usado para terminar um {\it loop} infinito.
 \item Por exemplo, podemos interromper uma contagem antes de seu final.
\end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Comando Break}
 {\bf Exemplo:}
\begin{lstlisting}
#include<stdio.h>
int main() {
  int i, n;
  for (i = 1; i <= 10; i++) {
      printf("%d ",i);
      if (i==5){
        printf("Contagem cancelada!");
        break;
      }
  }
  return 0;
}
\end{lstlisting}
Saída: 1 2 3 4 5 Contagem cancelada!
\end{frame}


\begin{frame}[fragile]
  \frametitle{Comando Break}

#include<stdio.h>

\section{Comando Continue}

\begin{frame}[fragile]
  \frametitle{Comando Continue}
\begin{itemize}
 \item O comando {\bf continue} faz com que uma iteração seja \underline{saltada}.
 \item É como se o fim do bloco tivesse sido alcançado
 \item O programa salta para a próxima iteração.
 \item Por exemplo, saltaremos o número 5 na contagem regressiva.
\end{itemize}
\end{frame}


\begin{frame}[fragile]
  \frametitle{Comando Continue}
 {\bf Exemplo:}
\begin{lstlisting}
#include<stdio.h>
int main() {
  int i, n;
  for (i = 1; i <= 10; i++) 
      if (i==5){
        continue;
      }
      printf("%d ",i);      
  }
  return 0;
}
\end{lstlisting}
Saída: 1 2 3 4 6 7 8 9 10
\end{frame}


\begin{frame}[fragile]
  \frametitle{Comando Continue}
 {\bf Exemplo:} programa que, dado o valor de $n$ fornecido pelo usuário, informe todos os divisores de $n$.
\end{frame}

\begin{frame}[fragile]
  \frametitle{Comando Continue}
Solução:
\begin{lstlisting}
#include<stdio.h>
int main() {
  int n, div;
  printf("Informe o valor de n:");
  scanf("%d",&n);
  printf("Os divisores de %d sao: ", n);
  for (div = 1; div < n; div++) {
      if (n%div != 0) {   /* Testa se nao eh divisor */
          continue;
      }
      printf("%d ",div); // Imprime o divisor
  }
  printf("%d\n.", n); /* n eh o ultimo divisor */
  return 0;
}
\end{lstlisting}
\end{frame}
