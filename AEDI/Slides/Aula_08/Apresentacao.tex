\section{Fluxogramas}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}
  \frametitle{Fluxogramas}
\begin{itemize}
\item Os fluxogramas são representações gráficas dos programas.
\item São utilizados para nos ajudar a compreender um programa.
\item Não estão associados a um linguagem específica.
\item Apresentam a lógica do algoritmo e não as instruções da linguagem.
\item Utilizam diferentes tipos de blocos para indicar os comandos (entradas, saídas, processamentos, decisões, etc) e setas para indicar
a sequência de execução.
\end{itemize}
\end{frame}

\section{Biblioteca \texttt{<math.h>}}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]{Biblioteca \texttt{<math.h>}}
\begin{itemize}
\item Como calcular $\pi r^2$?
\begin{itemize}
\item \texttt{double area = PI * raio * raio;}
\end{itemize}
\item A linguagem C não possui um operador para potência, mas possui uma biblioteca com diversas funções matemáticas, para usá-la
devemos incluir a biblioteca \texttt{math.h}
\item A função para potência é a \texttt{pow()}, sintaxe: 
\begin{itemize}
\item  \texttt{double pow(double base, double expoente);}
\end{itemize}
\item Exemplo
  \begin{lstlisting}
  // Programa que calcula a area de um circulo
  #include<math.h>
  ...
  double area = PI * pow(raio, 2);
  ...
  \end{lstlisting}  
\end{itemize}
\end{frame}

\section{Operadores Relacionais}

\begin{frame}{Operadores Relacionais}
\begin{itemize}
\item Permite a um programa realizar uma ação alternativa, a partir de um resultado {\color{blue}verdadeiro} ou {\color{red}falso} produzido por uma condição.
\item As {\bf condições} são formadas utilizando-se os operadores de igualdade e os operadores relacionais.
\item Ambos operadores de igualdade têm o mesmo nível de precedência, o qual é inferior ao dos operadores relacionais, e associam-se da esquerda para a direita.
\end{itemize}

\section{Operadores Lógicos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Operadores Lógicos}
\begin{itemize}
\item No século 18, George Boole, matemático e filósofo britânico, desenvolveu um sistema algébrico lógico, que passou a ser conhecido
como Álgebra de Boole.
\item Base para a lógica dos computadores digitais modernos.
\item Expressões lógicas (expressões booleanas) possuem o valor {\color{blue}true} ou {\color{red}false}.
\item Em C os inteiros também são usados como valores booleanos:
\begin{itemize}
\item qualquer valor não nulo ({\color{blue}1}) representa {\color{blue}true} e {\color{red}0} representa {\color{red}false}.
\end{itemize}
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Operadores Lógicos}{Expressões booleanas}
\begin{itemize}
\item na matemática, podemos restringir uma temperatura a um intervalo fechado, $0 \leq temp \leq 100$
\item em C não podemos representar essa expressão por $0 <= temp <= 100$.
\item A expressão $0 <= temp <= 100$ é uma expressão válida na linguagem C.
\begin{itemize}
\item no entanto, o resultado não será o esperado.
\end{itemize}
\item por exemplo, suponha que \texttt{temp = 150;} 
\begin{itemize}
\item não está no intervalo definido, logo esperamos que o resultado da expressão seja falso.
\end{itemize} 
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Operadores Lógicos}{Expressões booleanas}
\begin{itemize}
\item Os operadores relacionais são associativos a esquerda. A expressão será processada da seguinte forma:
\begin{figure}[!ht]
  \centering
  \includegraphics[width=90pt]{imgs/operacao_booleana_composta_ex1.png}
\end{figure}

\item A expressão (a) resulta em {\color{blue}true}, que é representado em C pelo inteiro 1 (ou outro inteiro diferente de 0).
\item Assim, na segunda etapa da avaliação, a expressão resulta em:

\begin{figure}[!ht]
  \centering
  \includegraphics[width=60pt]{imgs/operacao_booleana_composta_ex2.png}
\end{figure}

\item que também é verdadeira e resulta em {\color{blue}true}.
\item Entretanto, a expressão original deveria resultar {\color{red}false}, como na matemática.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Operadores Lógicos}{Expressões booleanas}
\begin{itemize}
\item Para solucionar esse problema, reescrevemos a desigualdade como:
\begin{equation}
(temp >= 0) \&\& (temp <= 100)\nonumber
\end{equation}
\item onde {\color{red}\&\&} é um operador lógico.
\item Utilizamos os operadores lógicos para combinar expressões booleanas formando, assim, expressões booleanas compostas
\end{itemize}
\end{frame}

\section{Comandos de Decisão}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Comandos de Decisão}
\begin{itemize}
\item Permite a um programa realizar uma ação alternativa, a partir de um resultado {\bf verdadeiro} ou {\bf falso} produzido por uma condição.
\item As condições são formadas utilizando-se os operadores de igualdade e os operadores relacionais.
\item Ambos operadores de igualdade têm o mesmo nível de precedência, o qual é inferior ao dos operadores relacionais, e associam-se da esquerda para a direita.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}{Comandos de Decisão}
Comando {\color{red}if}:
\begin{itemize}
\item consiste de uma palavra-chave {\color{red}if} seguida de uma expressão de teste entre parênteses. A instrução será executada apenas se a expressão de teste for {\color{blue}verdadeira}.
\item O corpo de um comando {\color{red}if} pode conter uma única instrução terminada por {\bf ponto-e-vírgula} ou várias instruções entre
{\bf chaves}.
\end{itemize}
\end{frame}

\begin{frame}{Comandos de Decisão}{Exemplo 1}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]{Fluxogramas}{Exemplo 2}
  \begin{lstlisting}
  // Programa que calcula a area de um circulo
  #include<stdio.h>
  int main() {
    // declaracao da constante Pi
    const double PI = 3.141592;
    double raio;
    
    printf("Digite o raio do circulo: ");
    scanf("%lf", &raio);
    
    // calculando e imprimindo a area
    if (raio >= 0) {
	    double area = PI * raio * raio;
	    printf("\nArea do circulo: %lf\n", area);
	}
    return 0;
  }
  \end{lstlisting}  
\end{frame}
